package com.opsflow.org_service.infrastructure.security;

import com.opsflow.common.JwtUtils;
import org.springframework.security.core.Authentication;
import org.springframework.security.core.context.SecurityContextHolder;
import org.springframework.stereotype.Service;

import java.util.Objects;

@Service("securityService")
public class SecurityService {

    private final JwtUtils jwtUtils;

    public SecurityService(JwtUtils jwtUtils) {
        this.jwtUtils = jwtUtils;
    }

    public boolean isMemberOfOrganization(Long organizationId) {
        Authentication auth = SecurityContextHolder.getContext().getAuthentication();
        if (auth == null || auth.getCredentials() == null) return false;

        String token = (String) auth.getCredentials();
        Long userOrgId = jwtUtils.getOrganizationIdFromToken(token);

        return Objects.equals(userOrgId, organizationId);
    }
}
