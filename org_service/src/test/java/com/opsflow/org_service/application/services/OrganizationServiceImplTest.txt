package com.opsflow.org_service.application.services;

import com.opsflow.org_service.application.dtos.OrganizationResponse;
import com.opsflow.org_service.application.dtos.request.OrganizationRequest;
import com.opsflow.org_service.domain.models.OrganizationDomain;
import com.opsflow.org_service.domain.ports.out.OrganizationRepositoryPort;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.ArgumentCaptor;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;

import java.time.LocalDateTime;
import java.util.List;
import java.util.Optional;

import static org.hibernate.validator.internal.util.Contracts.assertNotNull;
import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.ArgumentMatchers.any;
import static org.mockito.Mockito.*;

@ExtendWith(MockitoExtension.class)
class OrganizationServiceImplTest {
    @Mock
    private OrganizationRepositoryPort organizationRepositoryPort;

    @InjectMocks
    private OrganizationServiceImpl organizationService;

    private OrganizationRequest dummyRequest;
    private OrganizationDomain dummyDomain;

    @BeforeEach
    void setUp() {
        dummyRequest = new OrganizationRequest(
                "OpsFlow Inc",
                "123456789",
                "Calle Falsa 123",
                "contacto@opsflow.com",
                "555-1234",10);

        dummyDomain = new OrganizationDomain();
        dummyDomain.setId(1L);
        dummyDomain.setName("OpsFlow Inc");
        dummyDomain.setTaxId("123456789");
        dummyDomain.setAddress("Calle Falsa 123");
        dummyDomain.setEmail("contacto@opsflow.com");
        dummyDomain.setPhone("555-1234");
        dummyDomain.setActive(true);
        dummyDomain.setCreatedAt(LocalDateTime.now());
    }

    @Test
    void shouldCreate() {
        // GIVEN
        when(organizationRepositoryPort.save(any(OrganizationDomain.class))).thenReturn(dummyDomain);

        // WHEN
        OrganizationResponse response = organizationService.create(dummyRequest);

        // THEN
        assertNotNull(response);
        assertEquals(1L, response.id());
        assertEquals("OpsFlow Inc", response.name());
        assertTrue(response.active());

        ArgumentCaptor<OrganizationDomain> domainCaptor = ArgumentCaptor.forClass(OrganizationDomain.class);
        verify(organizationRepositoryPort, times(1)).save(domainCaptor.capture());

        OrganizationDomain capturedDomain = domainCaptor.getValue();
        assertEquals("OpsFlow Inc", capturedDomain.getName());
        assertTrue(capturedDomain.getActive());
        assertNotNull(capturedDomain.getCreatedAt());
    }

    @Test
    void shouldReturnOptionalWithResponse() {
        // GIVEN
        Long orgId = 1L;
        when(organizationRepositoryPort.findById(orgId)).thenReturn(Optional.of(dummyDomain));

        // WHEN
        Optional<OrganizationResponse> result = organizationService.findById(orgId);

        // THEN
        assertTrue(result.isPresent());
        assertEquals("OpsFlow Inc", result.get().name());
        verify(organizationRepositoryPort, times(1)).findById(orgId);
    }

    @Test
    void shouldReturnEmptyOptional() {
        // GIVEN
        Long orgId = 99L;
        when(organizationRepositoryPort.findById(orgId)).thenReturn(Optional.empty());

        // WHEN
        Optional<OrganizationResponse> result = organizationService.findById(orgId);

        // THEN
        assertTrue(result.isEmpty());
        verify(organizationRepositoryPort, times(1)).findById(orgId);
    }

    @Test
    void shouldReturnListOfOrganizationResponses() {
        // GIVEN
        when(organizationRepositoryPort.findAll()).thenReturn(List.of(dummyDomain));

        // WHEN
        List<OrganizationResponse> result = organizationService.findAll();

        // THEN
        assertNotNull(result);
        assertFalse(result.isEmpty());
        assertEquals(1, result.size());
        assertEquals("OpsFlow Inc", result.get(0).name());
        verify(organizationRepositoryPort, times(1)).findAll();
    }

    @Test
    void shouldUpdateOrganizationExists() {
        // GIVEN
        Long orgId = 1L;
        OrganizationRequest updateRequest = new OrganizationRequest(
                "OpsFlow Nuevo",
                "987654321",
                "Nueva Dirección",
                "nuevo@ops.com",
                "555-9999",
                10);

        when(organizationRepositoryPort.findById(orgId)).thenReturn(Optional.of(dummyDomain));
        when(organizationRepositoryPort.save(any(OrganizationDomain.class)))
                .thenAnswer(invocation -> invocation.getArgument(0));

        // WHEN
        Optional<OrganizationResponse> result = organizationService.update(orgId, updateRequest);

        // THEN
        assertTrue(result.isPresent());
        assertEquals("OpsFlow Nuevo", result.get().name());
        assertEquals("987654321", result.get().taxId());

        verify(organizationRepositoryPort, times(1)).findById(orgId);
        verify(organizationRepositoryPort, times(1)).save(any(OrganizationDomain.class));
    }

    @Test
    void shouldDeleteById() {
        // GIVEN
        Long orgId = 1L;
        doNothing().when(organizationRepositoryPort).deleteById(orgId);

        // WHEN
        organizationService.delete(orgId);

        // THEN
        verify(organizationRepositoryPort, times(1)).deleteById(orgId);
    }
}
